gamestat_df = []
with codecs.open(steamPageSaved, 'r',  encoding='utf8') as steamHTML:

    soup = BeautifulSoup(steamHTML.read(), 'lxml')
    # get title and hours played
    for game in soup.find_all('div', {'class': 'gameListRow'}):

        gameTitle = game.find('div', {'class': 'gameListRowItemName ellipsis '}).text
        hoursPlayed = game.find('h5', {'class': 'ellipsis hours_played'}).text.replace('hrs on record', '')
        if len(hoursPlayed) > 1:
            row = {
                'Title': gameTitle,
                'Hours': hoursPlayed.strip()
            }
            gamestat_df.append(row)

            
# add non-steam favourites
row = {
    'Title': 'Battlefield 1',
    'Hours': '72'
}
gamestat_df.append(row)
row = {
    'Title': 'Rainbow 6: Vegas 2',
    'Hours': '40'
}
gamestat_df.append(row)
            
# convert into a dataframe
gamestat_df = pd.DataFrame(gamestat_df)
